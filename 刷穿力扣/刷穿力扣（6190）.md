            if (len + words[i].length() + i - idx > maxWidth) {

                    sb.append(words[idx]);

            else l = mid + 1;
        }
        return (int)l - 1;
    }

class Solution {
    public int climbStairs(int n) {
        int l = 1, r = 1, cnt;
        for(int i = 0; i < n - 1; i++){
